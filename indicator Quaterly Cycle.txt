RemainingTime(_TF) =>
    if barstate.isrealtime
        timeRemaining = (_TF - timenow) / 1000
        days = math.floor(timeRemaining / 86400)
        hours = math.floor((timeRemaining - days * 86400) / 3600)
        minutes = math.floor((timeRemaining - days * 86400 - hours * 3600) / 60)
        seconds = math.floor(timeRemaining - days * 86400 - hours * 3600 - minutes * 60)

        r = str.tostring(seconds, '00')
        if minutes == 0
            r:= str.tostring(minutes, '00') + ':' + r
            r
        if minutes > 0 or hours > 0 or days > 0
            r := str.tostring(minutes, '00') + ':' + r
            r
        if hours > 0 or days > 0
            r := str.tostring(hours, '00') + ':' + r
            r
        if days > 0
            r := str.tostring(days) + 'D ' + r
            r
        r

    else
        'Market closed'

type dfr_item
    line[] lines
    box[] boxes
    label lbl

get_dfr(session, session_name, session_css, show, nn) =>
    var dfr_item[] history = array.new<dfr_item>()
    var t = 0 
    var max = high
    var min = low
    var line bxh = na
    var line bxl = na
    var label lbl = na
    
    // Arrays to manage the 8 levels efficiently (reduces tokens)
    var box[] active_boxes = array.new_box(0)
    var float[] active_levels = array.new_float(0)

    show_txt = show
            
    if session and not session[1]
        t := time
        max := high
        min := low
        
        bxh := line.new(t, max, t, max, xloc=xloc.bar_time, color=session_css, style=line.style_dotted, width=2)
        bxl := line.new(t, min, t, min, xloc=xloc.bar_time, color=session_css, style=line.style_dotted, width=2) 

        if show_txt
            lbl := label.new(t, max, session_name, xloc=xloc.bar_time, textcolor=session_css, style=label.style_label_down, color=color.new(color.white, 100), size=i_font)
        // Store in history
        item = dfr_item.new(array.new_line(), array.new_box(), lbl)
        array.push(item.lines, bxh)
        array.push(item.lines, bxl)
        array.push(history, item)
        // 4. Initialize the boxes for the new session
        array.clear(active_boxes)
        array.clear(active_levels)


        // STRICT MANAGEMENT: 
        // Use a 'while' loop instead of 'if' to force the size to match dfr_cycles
        while array.size(history) > dfr_cycles
            old = array.shift(history)
            for l in old.lines
                line.delete(l)
            for b in old.boxes
                box.delete(b)
            label.delete(old.lbl)

    if session
        max := math.max(high, max)
        min := math.min(low, min)
        line.set_xy1(bxh, t, max)
        line.set_xy2(bxh, time, max)
        line.set_xy1(bxl, t, min)
        line.set_xy2(bxl, time, min)

        if show_txt 
            label.set_xy(lbl, int(math.avg(t, time)), max)
            
        dr = max - min
        
        // Define levels in an array to loop through them
        levels = array.from(max + dr, max + 2 * dr, max + 2.5 * dr, max + 4 * dr, min - dr, min - 2 * dr, min - 2.5 * dr, min - 4 * dr)
        texts  = array.from("1", "2", "2.5", "4", "-1", "-2", "-2.5", "-4")

        if not session[1]
            // NEW SESSION: Create boxes and store in arrays
            array.clear(active_boxes)
            array.clear(active_levels)
            cur_item = array.get(history, array.size(history) - 1)
            
            for i = 0 to 7
                lvl_price = array.get(levels, i)
                txt = array.get(texts, i)
                v_align = i < 4 ? text.align_bottom : text.align_top
                
                bx = box.new(t, lvl_price, time, lvl_price, 
                     xloc=xloc.bar_time, 
                     border_color=session_css, 
                     bgcolor=color.new(color.red, 100), 
                     text=txt, 
                     text_color=session_css, 
                     text_valign=v_align, 
                     text_halign=text.align_right, 
                     text_size=size.tiny, 
                     border_style=line.style_dotted)
                
                array.push(active_boxes, bx)
                array.push(active_levels, lvl_price)

f_cycle_box_manage(new_cycle, lbl_array, val_index)=>
    var box[] boxes = array.new_box()
    
    lbl = array.get(lbl_array, val_index)
    q = f_q_lbl(lbl)
    
    c_color = q == "Q1" ? q1_box_clr : q == "Q2" ? q2_box_clr : q == "Q3" ? q3_box_clr : q == "Q4" ? q4_box_clr : q1_box_clr
    
    transp = canvas_theme == "Dark" ? 85 : 90
    bg_color = color.new(c_color, transp)
    border_color = color.new(c_color, 50)
    
    if new_cycle
        if array.size(boxes) > 0
            prev_b = array.get(boxes, 0)
            box.set_right(prev_b, time)
        b = box.new(left=time, top=high, right=time, bottom=low, xloc=xloc.bar_time, border_color=border_color, border_style=line.style_dashed, bgcolor=bg_color)
        array.unshift(boxes, b)
        if show_last and array.size(boxes) > num_cycle
            box.delete(array.pop(boxes))

    if show_today_only
        t_day = time("1D")
        if not na(t_day)
            while array.size(boxes) > 0
                last_b = array.get(boxes, array.size(boxes) - 1)
                if box.get_left(last_b) < t_day
                    box.delete(array.pop(boxes))
                else
                    break
            
    if array.size(boxes) > 0
        b = array.get(boxes, 0)
        box.set_right(b, time)
        if high > box.get_top(b)
            box.set_top(b, high)
        if low < box.get_bottom(b)
            box.set_bottom(b, low)

type SessionState
    float h = na
    int ht = na
    float l = na
    int lt = na
    bool hs = false
    int hst = na
    bool ls = false
    int lst = na
    float cur_h = high
    int cur_ht = time
    float cur_l = low
    int cur_lt = time
    box bh = na
    box bl = na

f_update_htf_state(SessionState s, bool new_period) =>
    if new_period
        s.h := s.cur_h, s.ht := s.cur_ht
        s.l := s.cur_l, s.lt := s.cur_lt
        s.cur_h := high, s.cur_ht := time
        s.cur_l := low, s.cur_lt := time
        s.hs := false, s.ls := false
    else
        if high > s.cur_h
            s.cur_h := high, s.cur_ht := time
        if low < s.cur_l
            s.cur_l := low, s.cur_lt := time
            
    if not s.hs and not na(s.h) and high > s.h
        s.hs := true, s.hst := time
    if not s.ls and not na(s.l) and low < s.l
        s.ls := true, s.lst := time
    0

f_update_session_state(SessionState s, bool active) =>
    if active
        if not active[1]
            s.h := high, s.ht := time
            s.l := low, s.lt := time
            s.hs := false, s.ls := false
        else
            if high > s.h
                s.h := high, s.ht := time
            if low < s.l
                s.l := low, s.lt := time
    else
        if not s.hs and not na(s.h) and high > s.h
            s.hs := true, s.hst := time
        if not s.ls and not na(s.l) and low < s.l
            s.ls := true, s.lst := time
    0

f_render_state(SessionState s, bool show, string txt_h, string txt_l, color c, int dt) =>
    if show and not na(s.h)
        s.bh := f_manage_level_box(s.bh, s.ht, s.h, s.hs ? s.hst : time + 15*dt, txt_h, c)
        s.bl := f_manage_level_box(s.bl, s.lt, s.l, s.ls ? s.lst : time + 15*dt, txt_l, c)
    else
        box.delete(s.bh), s.bh := na
        box.delete(s.bl), s.bl := na

f_nwox_main(_tf, clr, del, _show)=>
    // Arrays to store drawing objects
    var nwox_top_lines  = array.new_line()
    var nwox_bot_lines  = array.new_line()
    var nwox_ce_lines   = array.new_line()
    var nwox_link_lines = array.new_line() 
    var nwox_fills      = array.new_linefill()
    var nwox_labels     = array.new_label()

    // Manual mode state
    var bool manual_gap_found = false
    var int manual_gap_time = na

    var float csd_level = na
    _tap = false 
    htapW = false, htapD = false, htap90 = false, htapm = false
    ltapW = false, ltapD = false, ltap90 = false, ltapm = false    

    // Calculate Extension Buffer (Same as 1m FVG)
    // Uses the global 'settings.buffer' defined earlier in your script
    int tc = (time - time[1]) * settings.buffer

    // --- Detect and Process New Gap on Timeframe Change ---
    if timeframe.change(_tf) 
        // Calculate Gap Levels
        _max = math.max(close[1], open)
        _min = math.min(close[1], open)
        _ce  = math.avg(_max, _min)
        
        // Determine Colors
        gap_border_clr = i_textcolor 
        gap_fill_clr   = color.new(clr, 90) 

        // --- NWOG Manual Mode Logic ---
        bool is_manual_target = false
        if _tf == "1W" and nwog_mode == "Manual" and _show
            int target_month = f_month_from_string(nwog_manual_month)
            int target_quarter = f_quarter_from_string(nwog_manual_quarter)
            int wom = math.ceil(dayofmonth(time) / 7.0)
            if month(time) == target_month and wom == target_quarter
                is_manual_target := true

            // If we find the manual target, store it and prevent other gaps from drawing
            if is_manual_target
                manual_gap_found := true
                manual_gap_time := time

        // --- Drawing Condition ---
        bool can_draw = false
        if _show and _max - _min >= 1 * syminfo.mintick and not (_tf == "1D" and timeframe.change("1W"))
            if _tf == "1W"
                if nwog_mode == "Auto"
                    can_draw := true
                else if is_manual_target // Manual mode, only draw the specific target
                    can_draw := true
            else if _tf == "1D" // Old NDOG logic
                can_draw := true

        if can_draw
            
            
            // 1. Create Horizontal Lines (Extended to time + tc)
            l_top = line.new(x1=time, y1=_max, x2=time + tc, y2=_max, xloc=xloc.bar_time, color=gap_border_clr, style=line.style_solid)
            l_bot = line.new(x1=time, y1=_min, x2=time + tc, y2=_min, xloc=xloc.bar_time, color=gap_border_clr, style=line.style_solid)
            l_ce  = line.new(x1=time, y1=_ce,  x2=time + tc, y2=_ce,  xloc=xloc.bar_time, color=gap_border_clr, style=line.style_dashed)
            l_fill = linefill.new(l_top, l_bot, gap_fill_clr)

            // 2. Create Vertical Link Line (At the end of the extension)
            l_link = line.new(x1=time + tc, y1=_max, x2=time + tc, y2=_min, xloc=xloc.bar_time, color=gap_border_clr, style=line.style_solid, width=3)

            // 3. Generate Label Text
            string lbl_txt = ""
            if _tf == "1W" // NWOG Logic
                int wom = math.ceil(dayofmonth(time) / 7.0)
                lbl_txt := str.format("{0,date,MMMM} Q{1} NWOG", time, wom)
            else if _tf == "1D" // NDOG Logic
                lbl_txt := str.format("{0,date,EEE dd-MM-yyyy} NDOG", time)

            // 4. Create Label (At the end of the extension)
            l_lbl = label.new(x=time + tc, y=_ce, text=lbl_txt, xloc=xloc.bar_time, textcolor=gap_border_clr, style=label.style_label_left, color=color.new(color.white, 100), size=size.small)

            // Store in arrays
            array.unshift(nwox_top_lines, l_top)
            array.unshift(nwox_bot_lines, l_bot)
            array.unshift(nwox_ce_lines, l_ce)
            array.unshift(nwox_link_lines, l_link)
            array.unshift(nwox_fills, l_fill)
            array.unshift(nwox_labels, l_lbl)
            
            // Maintain Array Size Limit
            if array.size(nwox_top_lines) > 60 // Keep a generous buffer for lookback
                line.delete(array.pop(nwox_top_lines))
                line.delete(array.pop(nwox_bot_lines))
                line.delete(array.pop(nwox_ce_lines))
                line.delete(array.pop(nwox_link_lines))
                linefill.delete(array.pop(nwox_fills))
                label.delete(array.pop(nwox_labels))

    // --- Logic for Extending Lines, Moving Labels/Links, and Detecting Taps ---
    dxx   = ta.atr(20)/2
    hmax  = array.get(N_ext,0), lmin  = array.get(N_ext,1)
    hmaxD = array.get(D_ext,0), lminD = array.get(D_ext,1)
    hmaxW = array.get(W_ext,0), lminW = array.get(W_ext,1)
    hmaxm = array.get(m_ext,0), lminm = array.get(m_ext,1)    

    if array.size(nwox_top_lines) > 0
        for i = array.size(nwox_top_lines)-1 to 0
            l_top  = array.get(nwox_top_lines, i)
            l_bot  = array.get(nwox_bot_lines, i)
            l_ce   = array.get(nwox_ce_lines, i)
            l_link = array.get(nwox_link_lines, i)
            l_lbl  = array.get(nwox_labels, i)
            
            top = line.get_y1(l_top)
            bot = line.get_y1(l_bot)
            
            // Extend Horizontal Lines to current bar + buffer (tc)
            line.set_x2(l_top, time + tc)
            line.set_x2(l_bot, time + tc)
            line.set_x2(l_ce,  time + tc)
            
            // Move Vertical Link Line to current bar + buffer (tc)
            line.set_x1(l_link, time + tc)
            line.set_x2(l_link, time + tc)

            // Move Label to current bar + buffer (tc)
            label.set_x(l_lbl, time + tc)

            // Tap Logic
            if math.abs(bot - high) <= dxx or math.abs(low - top) <= dxx or (high <= top and low >= bot) or (high >= bot and low <= top)
                _tap := true
            if ((hmax > top and lmin < top + dxx) or ( top >= hmax and bot <= lmin)) and tf_ok 
                ltap90 := true
            if ((lmin < bot and hmax > bot - dxx) or ( top >= hmax and bot <= lmin)) and tf_ok 
                htap90 := true
            if ((hmaxW > top and lminW < top + dxx) or ( top >= hmaxW and bot <= lminW)) and tf_okW 
                ltapW := true
            if ((lminW < bot and hmaxW > bot - dxx) or ( top >= hmaxW and bot <= lminW)) and tf_okW 
                htapW := true  
            if ((hmaxD > top and lminD < top + dxx) or ( top >= hmaxD and bot <= lminD)) and tf_okD 
                ltapD := true
            if ((lminD < bot and hmaxD > bot - dxx) or ( top >= hmaxD and bot <= lminD)) and tf_okD 
                htapD := true                 
            if ((hmaxm > top and lminm < top + dxx) or ( top >= hmaxm and bot <= lminm)) and tf_ok_mic 
                ltapm := true
            if ((lminm < bot and hmaxm > bot - dxx) or ( top >= hmaxm and bot <= lminm)) and tf_ok_mic 
                htapm := true

    // --- Cleanup Old Objects ---
    // In Auto mode, limit by number. In Manual, we only ever have one.
    int final_del = del
    if _tf == "1W"
        final_del := nwog_mode == "Auto" ? nwog_num_auto : 1

    if array.size(nwox_top_lines) > final_del
        for i = array.size(nwox_top_lines)-1 to final_del
            // Check if we are in manual mode and the line to be deleted is the one we want to keep
            if _tf == "1W" and manual_gap_found and line.get_x1(array.get(nwox_top_lines, i)) == manual_gap_time
                continue

            line.delete(array.pop(nwox_top_lines))
            line.delete(array.pop(nwox_bot_lines))
            line.delete(array.pop(nwox_ce_lines))
            line.delete(array.pop(nwox_link_lines))
            linefill.delete(array.pop(nwox_fills))
            label.delete(array.pop(nwox_labels))

    // Return values
    [_tap, ltapW, htapW, ltapD, htapD, ltap90, htap90, ltapm, htapm]

f_nwox_corele(_tf, fc1, fo0, fh0, fl0, del, fatr, gc) =>
    var nwox_max = array.new_float()
    var nwox_min = array.new_float()
    _tap = false
    
    // Detect New Gap
    if timeframe.change(_tf) 
        _max = math.max(fc1, fo0) // fc1 is close[1] of the correlated symbol
        _min = math.min(fc1, fo0)
        // Only store if gap exists
        if _max - _min >= 1 * syminfo.mintick and not (_tf == "1D" and timeframe.change("1W") and show_nwog)
            // Use generic array push if add_to_array_box isn't working for floats, 
            // or keep your helper function if it handles generics.
            // Safe fallback using standard array functions:
            array.unshift(nwox_max, _max)
            if array.size(nwox_max) > 40
                array.pop(nwox_max)
                
            array.unshift(nwox_min, _min)
            if array.size(nwox_min) > 40
                array.pop(nwox_min)

    hmax = array.get(N_ext, 0)
    lmin = array.get(N_ext, 1)  

    dxx = fatr / 2

    if array.size(nwox_max) > 0
        for i = array.size(nwox_max) - 1 to 0
            top = array.get(nwox_max, i)
            bot = array.get(nwox_min, i)
            
            // Logic to detect if price tapped the gap
            if gc == "c" and (math.abs(bot - fh0) <= dxx or math.abs(fl0 - top) <= dxx or (fh0 <= top and fl0 >= bot) or (fh0 >= bot and fl0 <= top))
                _tap := true
            if gc == "l" and ((fh0 > top and fl0 < top + dxx) or (top >= fh0 and bot <= fl0))  
                _tap := true
            if gc == "h" and ((lmin < bot and hmax > bot - dxx) or (top >= hmax and bot <= lmin))  
                _tap := true            

    // Cleanup old values
    if array.size(nwox_max) > del
        for i = array.size(nwox_max) - 1 to del
            array.remove(nwox_max, i)
            array.remove(nwox_min, i)

    _tap

method updateFill(OpenRange this) =>
    if this.open < this.close
        if this.box_fill2.get_bottom() >= this.open and low < this.box_fill2.get_bottom()
            this.box_fill2.set_bottom(math.max(low, this.open))
            this.box_fill1.set_bottom(this.box_fill2.get_bottom())
    else
        if this.box_fill2.get_bottom() <= this.open and high > this.box_fill2.get_bottom()
            this.box_fill2.set_bottom(math.min(high, this.open))
            this.box_fill1.set_bottom(this.box_fill2.get_bottom())

                        label_text = "1st FVG " + date_str
                        org.label_fvg := label.new(time + tc, (o + c) / 2, label_text, xloc = xloc.bar_time, textcolor = fvg_color, color = settings.fvg_label_bg, size = settings.fvg_label_size, style = label.style_label_left)
                        org.label_fvg
                    if settings.show_fvg_bg
                        org.lf_fvg := linefill.new(org.line_fvg_o, org.line_fvg_c, fvg_bg_color)
                        org.lf_fvg
    
    int max_history = math.max(settings.max_fvg, settings.max_org)
    if ranges.size() > settings.max_fvg
        for i = settings.max_fvg to ranges.size() - 1

    if ranges.size() > settings.max_org

// Helper function to check if a specific SSMT array has active signals
f_check_bear_ssmt(_arr) => 
    array.size(_arr) > 9 ? (array.get(_arr, 1) or array.get(_arr, 4) or array.get(_arr, 6) or array.get(_arr, 8)) : false

bool alert_on_creation = QCISD_alert_type == "All" or QCISD_alert_type == "Creation"

bool alert_on_retest = QCISD_alert_type == "All" or QCISD_alert_type == "Retest"

f_check_bull_ssmt(_arr) => 
    array.size(_arr) > 9 ? (array.get(_arr, 3) or array.get(_arr, 5) or array.get(_arr, 7) or array.get(_arr, 9)) : false

// GLOBAL PERSISTENT VARIABLES (Arrays for lines and labels)

var hp_csd_lbls = array.new_label()
// 2. CSD Function
f_run_csd(_buffer, _max_show) => 
    // State variables to hold the candle info that created the SSMT
    // We maintain High (h), Low (l), and Open (o) to handle the priority logic
    var float bull_ssmt_o = na
    var float bull_ssmt_h = na
    var int   bull_ssmt_t = na
    var string bull_ssmt_src = "" // NEW: To store which cycle created the signal
    var int bull_ssmt_idx = 0
    var string bull_ssmt_bias = "" // For HP-CISD
    
    var float bear_ssmt_o = na
    var float bear_ssmt_l = na
    var int   bear_ssmt_t = na
    var string bear_ssmt_src = "" // NEW: To store which cycle created the signal
    var int bear_ssmt_idx = 0
    var string bear_ssmt_bias = "" // For HP-CISD

    // 1. EXTENSION: 15 candles (Fixed length)
    // int ext_len = (time - time[1]) * 20
    int ext_len = timeframe.in_seconds(timeframe.period)*1000 * 20

    // 2. CHECK ACTIVE SSMT STATUS WITH STRICT TIMEFRAME/VISIBILITY VALIDATION (FORCED AUTO MODE)
    // Create dedicated visibility variables for CISD that always use the 'Auto' mode logic.
    bool cisd_tf_ok_mic_V = (timeframe.in_seconds(tf) >= timeframe.in_seconds(botm) and timeframe.in_seconds(tf) <= timeframe.in_seconds(topm))
    bool cisd_tf_ok_V = (timeframe.in_seconds(tf) >= timeframe.in_seconds(botN) and timeframe.in_seconds(tf) <= timeframe.in_seconds(topN))
    bool cisd_tf_okD_V = (timeframe.in_seconds(tf) >= timeframe.in_seconds(botD) and timeframe.in_seconds(tf) <= timeframe.in_seconds(topD))
    bool cisd_tf_okW_V = (timeframe.in_seconds(tf) >= timeframe.in_seconds(botW) and timeframe.in_seconds(tf) <= timeframe.in_seconds(topW))
    bool cisd_tf_okMo_V = (timeframe.in_seconds(tf) >= timeframe.in_seconds(botMo) and timeframe.in_seconds(tf) <= timeframe.in_seconds(topMo))

    
    // Micro Cycle
    bool valid_mic = show_cyclesmic and tf_ok_mic and cisd_tf_ok_mic_V
    bool is_bull_m = valid_mic and f_check_bull_ssmt(m_SSMT)
    bool is_bear_m = valid_mic and f_check_bear_ssmt(m_SSMT)

    // 90m Cycle
    bool valid_90 = show_cycles90 and tf_ok and cisd_tf_ok_V
    bool is_bull_n = valid_90 and f_check_bull_ssmt(N_SSMT)
    bool is_bear_n = valid_90 and f_check_bear_ssmt(N_SSMT)

    // Daily Cycle
    bool valid_D = show_cyclesD and tf_okD and cisd_tf_okD_V
    bool is_bull_d = valid_D and f_check_bull_ssmt(D_SSMT)
    bool is_bear_d = valid_D and f_check_bear_ssmt(D_SSMT)

    // Weekly Cycle
    bool valid_W = show_cyclesW and tf_okW and cisd_tf_okW_V
    bool is_bull_w = valid_W and f_check_bull_ssmt(W_SSMT)
    bool is_bear_w = valid_W and f_check_bear_ssmt(W_SSMT)

    // Monthly Cycle (Added)
    bool valid_Mo = show_cyclesMo and tf_okMo and cisd_tf_okMo_V
    bool is_bull_mo = valid_Mo and f_check_bull_ssmt(M_SSMT)
    bool is_bear_mo = valid_Mo and f_check_bear_ssmt(M_SSMT)

    // Combined Status
    bool is_bull_any = is_bull_m or is_bull_n or is_bull_d or is_bull_w or is_bull_mo
    bool is_bear_any = is_bear_m or is_bear_n or is_bear_d or is_bear_w or is_bear_mo

    // DETECT RISING EDGE (New Signal Only)
    bool new_bull_ssmt = is_bull_any and not is_bull_any[1]
    bool new_bear_ssmt = is_bear_any and not is_bear_any[1]

    // 3. CAPTURE LEVELS AND SOURCE TAG (Only on New Signal)
    if new_bull_ssmt
        // Determine Source Tag based on priority (Smallest to Largest usually)
        if is_bull_m
            bull_ssmt_src := "Micro"
        else if is_bull_n
            bull_ssmt_src := "90M"
        else if is_bull_d
            bull_ssmt_src := "Daily"
        else if is_bull_w
            bull_ssmt_src := "Weekly"
        else if is_bull_mo
            bull_ssmt_src := "Monthly"

        bull_ssmt_idx := bar_index
        // Bullish SSMT Logic (Reference Capture)
        bull_ssmt_o := na
        bull_ssmt_h := na
        for i = 0 to 200
            if close[i] < open[i]
                bull_ssmt_o := open[i]
                bull_ssmt_t := time[i]
                break
        
        // Reset the opposite signal
        if bar_index - bear_ssmt_idx > csd_relaxation
            bear_ssmt_o := na
            bear_ssmt_l := na
            bear_ssmt_src := ""

    if new_bear_ssmt
        // Determine Source Tag
        if is_bear_m
            bear_ssmt_src := "Micro"
        else if is_bear_n
            bear_ssmt_src := "90M"
        else if is_bear_d
            bear_ssmt_src := "Daily"
        else if is_bear_w
            bear_ssmt_src := "Weekly"
        else if is_bear_mo
            bear_ssmt_src := "Monthly"

        bear_ssmt_idx := bar_index
        // Bearish SSMT Logic (Reference Capture)
        bear_ssmt_o := na
        bear_ssmt_l := na
        for i = 0 to 200
            if close[i] > open[i]
                bear_ssmt_o := open[i]
                bear_ssmt_t := time[i]
                break
        
        // Reset the opposite signal
        if bar_index - bull_ssmt_idx > csd_relaxation
            bull_ssmt_o := na
            bull_ssmt_h := na
            bull_ssmt_src := ""

    // 4. EXECUTION: Check for Breakout (CISD) with Priority
    if show_csd and barstate.isconfirmed

        // --- BULLISH PLOTTING LOGIC ---
        // Priority 1: Check Open Logic
        if not na(bull_ssmt_o) and close > bull_ssmt_o
            float csd_level = bull_ssmt_o
            int csd_idx = bull_ssmt_t
            color csd_col = csd_bull_clr
            string final_csd_txt = "(+)" + bull_ssmt_src + "_QCISD"

            // --- High Probability (HP) CISD Logic with HTF Bias Alignment ---
            string required_bias = ""
            if bull_ssmt_src == "Micro"
                required_bias := array.get(D_bias, 0)
            else if bull_ssmt_src == "90M"
                required_bias := array.get(W_bias, 0)
            else if bull_ssmt_src == "Daily"
                required_bias := array.get(W_bias, 0)
            else if bull_ssmt_src == "Weekly"
                required_bias := array.get(M_bias, 0)

            if show_hp_csd and required_bias == "Bullish"
                // If HP condition is met, plot HP line and DO NOT plot normal CISD
                string hp_csd_txt = "(+HP)" + bull_ssmt_src + "_QCISD"
                line l_hp_csd = line.new(x1=csd_idx, y1=csd_level, x2=time + ext_len, y2=csd_level, xloc=xloc.bar_time, color=hp_csd_bull_clr, style=line.style_solid, width=3)
                label l_hp_lbl = label.new(x=time + ext_len, y=csd_level, text=hp_csd_txt, xloc=xloc.bar_time, style=label.style_label_left, color=color.new(color.white, 100), textcolor=i_textcolor, size=size.small)
                array.unshift(hp_csd_lines, l_hp_csd)
                array.unshift(hp_csd_lbls, l_hp_lbl)
                if alert_on_creation
                    alert(hp_csd_txt + " created at " + str.tostring(csd_level), alert.freq_once_per_bar_close)
            else
                // If HP condition is NOT met, plot the normal CISD line
                line l_csd = line.new(x1=csd_idx, y1=csd_level, x2=time + ext_len, y2=csd_level, xloc=xloc.bar_time, color=csd_col, style=line.style_solid, width=2)
                label l_lbl = label.new(x=time + ext_len, y=csd_level, text=final_csd_txt, xloc=xloc.bar_time, style=label.style_label_left, color=color.new(color.white, 100), textcolor=i_textcolor, size=size.small)
                array.unshift(csd_lines, l_csd)
                array.unshift(csd_lbls, l_lbl)
                if alert_on_creation
                    alert(final_csd_txt + " created at " + str.tostring(csd_level), alert.freq_once_per_bar_close)

            // Signal executed, reset state
            bull_ssmt_o := na
            bull_ssmt_h := na
            bull_ssmt_bias := ""
            bull_ssmt_src := ""

        // Priority 2: Check High Logic
        else if not na(bull_ssmt_h) and close > bull_ssmt_h
            float csd_level = bull_ssmt_h
            int csd_idx = bull_ssmt_t
            color csd_col = csd_bull_clr
            string final_csd_txt = "(+)" + bull_ssmt_src + "_QCISD"

            // --- High Probability (HP) CISD Logic with HTF Bias Alignment ---
            string required_bias = ""
            if bull_ssmt_src == "Micro"
                required_bias := array.get(D_bias, 0)
            else if bull_ssmt_src == "90M"
                required_bias := array.get(W_bias, 0)
            else if bull_ssmt_src == "Daily"
                required_bias := array.get(W_bias, 0)
            else if bull_ssmt_src == "Weekly"
                required_bias := array.get(M_bias, 0)

            if show_hp_csd and required_bias == "Bullish"
                // If HP condition is met, plot HP line
                string hp_csd_txt = "(+HP)" + bull_ssmt_src + "_QCISD"
                line l_hp_csd = line.new(x1=csd_idx, y1=csd_level, x2=time + ext_len, y2=csd_level, xloc=xloc.bar_time, color=hp_csd_bull_clr, style=line.style_solid, width=3)
                label l_hp_lbl = label.new(x=time + ext_len, y=csd_level, text=hp_csd_txt, xloc=xloc.bar_time, style=label.style_label_left, color=color.new(color.white, 100), textcolor=i_textcolor, size=size.small)
                array.unshift(hp_csd_lines, l_hp_csd)
                array.unshift(hp_csd_lbls, l_hp_lbl)
                if alert_on_creation
                    alert(hp_csd_txt + " created at " + str.tostring(csd_level), alert.freq_once_per_bar_close)
            else
                // If HP condition is NOT met, plot the normal CISD line
                line l_csd = line.new(x1=csd_idx, y1=csd_level, x2=time + ext_len, y2=csd_level, xloc=xloc.bar_time, color=csd_col, style=line.style_solid, width=2)
                label l_lbl = label.new(x=time + ext_len, y=csd_level, text=final_csd_txt, xloc=xloc.bar_time, style=label.style_label_left, color=color.new(color.white, 100), textcolor=i_textcolor, size=size.small)
                array.unshift(csd_lines, l_csd)
                array.unshift(csd_lbls, l_lbl)
                if alert_on_creation
                    alert(final_csd_txt + " created at " + str.tostring(csd_level), alert.freq_once_per_bar_close)

            bull_ssmt_o := na
            bull_ssmt_h := na
            bull_ssmt_bias := ""
            bull_ssmt_src := ""

        // --- BEARISH PLOTTING LOGIC ---
        // Priority 1: Check Open Logic
        if not na(bear_ssmt_o) and close < bear_ssmt_o
            float csd_level = bear_ssmt_o
            int csd_idx = bear_ssmt_t
            color csd_col = csd_bear_clr
            string final_csd_txt = "(-)" + bear_ssmt_src + "_QCISD"

            // --- High Probability (HP) CISD Logic with HTF Bias Alignment ---
            string required_bias = ""
            if bear_ssmt_src == "Micro"
                required_bias := array.get(D_bias, 0)
            else if bear_ssmt_src == "90M"
                required_bias := array.get(W_bias, 0)
            else if bear_ssmt_src == "Daily"
                required_bias := array.get(W_bias, 0)
            else if bear_ssmt_src == "Weekly"
                required_bias := array.get(M_bias, 0)

            if show_hp_csd and required_bias == "Bearish"
                // If HP condition is met, plot HP line
                string hp_csd_txt = "(-HP)" + bear_ssmt_src + "_QCISD"
                line l_hp_csd = line.new(x1=csd_idx, y1=csd_level, x2=time + ext_len, y2=csd_level, xloc=xloc.bar_time, color=hp_csd_bear_clr, style=line.style_solid, width=3)
                label l_hp_lbl = label.new(x=time + ext_len, y=csd_level, text=hp_csd_txt, xloc=xloc.bar_time, style=label.style_label_left, color=color.new(color.white, 100), textcolor=i_textcolor, size=size.small)
                array.unshift(hp_csd_lines, l_hp_csd)
                array.unshift(hp_csd_lbls, l_hp_lbl)
                if alert_on_creation
                    alert(hp_csd_txt + " created at " + str.tostring(csd_level), alert.freq_once_per_bar_close)
            else
                // If HP condition is NOT met, plot the normal CISD line
                line l_csd = line.new(x1=csd_idx, y1=csd_level, x2=time + ext_len, y2=csd_level, xloc=xloc.bar_time, color=csd_col, style=line.style_solid, width=2)
                label l_lbl = label.new(x=time + ext_len, y=csd_level, text=final_csd_txt, xloc=xloc.bar_time, style=label.style_label_left, color=color.new(color.white, 100), textcolor=i_textcolor, size=size.small)
                array.unshift(csd_lines, l_csd)
                array.unshift(csd_lbls, l_lbl)
                if alert_on_creation
                    alert(final_csd_txt + " created at " + str.tostring(csd_level), alert.freq_once_per_bar_close)

            bear_ssmt_o := na
            bear_ssmt_l := na
            bear_ssmt_bias := ""
            bear_ssmt_src := ""

        // Priority 2: Check Low Logic
        else if not na(bear_ssmt_l) and close < bear_ssmt_l
            float csd_level = bear_ssmt_l
            int csd_idx = bear_ssmt_t
            color csd_col = csd_bear_clr
            string final_csd_txt = "(-)" + bear_ssmt_src + "_QCISD"
            
            // --- High Probability (HP) CISD Logic with HTF Bias Alignment ---
            string required_bias = ""
            if bear_ssmt_src == "Micro"
                required_bias := array.get(D_bias, 0)
            else if bear_ssmt_src == "90M"
                required_bias := array.get(W_bias, 0)
            else if bear_ssmt_src == "Daily"
                required_bias := array.get(W_bias, 0)
            else if bear_ssmt_src == "Weekly"
                required_bias := array.get(M_bias, 0)

            if show_hp_csd and required_bias == "Bearish"
                // If HP condition is met, plot HP line
                string hp_csd_txt = "(-HP)" + bear_ssmt_src + "_QCISD"
                line l_hp_csd = line.new(x1=csd_idx, y1=csd_level, x2=time + ext_len, y2=csd_level, xloc=xloc.bar_time, color=hp_csd_bear_clr, style=line.style_solid, width=3)
                label l_hp_lbl = label.new(x=time + ext_len, y=csd_level, text=hp_csd_txt, xloc=xloc.bar_time, style=label.style_label_left, color=color.new(color.white, 100), textcolor=i_textcolor, size=size.small)
                array.unshift(hp_csd_lines, l_hp_csd)
                array.unshift(hp_csd_lbls, l_hp_lbl)
                if alert_on_creation
                    alert(hp_csd_txt + " created at " + str.tostring(csd_level), alert.freq_once_per_bar_close)
            else
                // If HP condition is NOT met, plot the normal CISD line
                line l_csd = line.new(x1=csd_idx, y1=csd_level, x2=time + ext_len, y2=csd_level, xloc=xloc.bar_time, color=csd_col, style=line.style_solid, width=2)
                label l_lbl = label.new(x=time + ext_len, y=csd_level, text=final_csd_txt, xloc=xloc.bar_time, style=label.style_label_left, color=color.new(color.white, 100), textcolor=i_textcolor, size=size.small)
                array.unshift(csd_lines, l_csd)
                array.unshift(csd_lbls, l_lbl)
                if alert_on_creation
                    alert(final_csd_txt + " created at " + str.tostring(csd_level), alert.freq_once_per_bar_close)

            bear_ssmt_o := na
            bear_ssmt_l := na
            bear_ssmt_bias := ""
            bear_ssmt_src := ""

        // --- Line Management (Max Lines Only) ---
        while array.size(csd_lines) > _max_show
            line.delete(array.pop(csd_lines))
            label.delete(array.pop(csd_lbls))
        while array.size(hp_csd_lines) > _max_show
            line.delete(array.pop(hp_csd_lines))
            label.delete(array.pop(hp_csd_lbls))
           
    // RELAXED MITIGATION: delete only if 3 consecutive body breaks (original + next 2)
    if array.size(csd_lines) > 0
        for i = array.size(csd_lines) - 1 to 0
            line  l_curr   = array.get(csd_lines, i)
            label lbl_curr = array.get(csd_lbls, i)

            float lvl = line.get_y1(l_curr)
            string txt = label.get_text(lbl_curr)

            bool delete_it = false


            // --------------------------
            // BULLISH CISD (+)
            // --------------------------
            if str.contains(txt, "(+)")
                bool body_below_0 = (close < lvl and open < lvl)       // current

                // --- Invalidation Logic ---
                bool body_below_1 = (close[1] < lvl and open[1] < lvl) // next candle
                bool body_below_2 = (close[2] < lvl and open[2] < lvl) // next-next candle

                // First break → pending color
                if body_below_0
                    line.set_color(l_curr, csd_pending_color)

                // Delete only if 3 breaks in a row
                if body_below_0 and body_below_1 and body_below_2
                    delete_it := true

                // Recovery → restore color
                if close > lvl or open > lvl
                    line.set_color(l_curr, color.green)

                // Retest & Rejection Alert Logic
                bool retest_bull = low[1] <= lvl and (close[1] < lvl or open[1] < lvl) // Previous bar touched/pierced
                bool reject_bull = close > lvl and open > lvl // Current bar closes back above
                if alert_on_retest and retest_bull and reject_bull
                    alert("Retest and Rejection of " + txt + " at " + str.tostring(lvl), alert.freq_once_per_bar_close)

            // --------------------------
            // BEARISH CISD (-)
            // --------------------------
            else if str.contains(txt, "(-)")
                // --- Invalidation Logic ---
                bool body_above_0 = (close > lvl and open > lvl)
                bool body_above_1 = (close[1] > lvl and open[1] > lvl)
                // bool body_above_2 = (close[2] > lvl and open[2] > lvl)

                // First break → pending color
                if body_above_0
                    line.set_color(l_curr, csd_pending_color)

                // Delete only if 3 breaks in a row
                if body_above_0 and body_above_1 // and body_above_2
                    delete_it := true

                // Recovery → restore color
                if close < lvl or open < lvl
                    line.set_color(l_curr, color.red)

                // Retest & Rejection Alert Logic
                bool retest_bear = high[1] >= lvl and (close[1] > lvl or open[1] > lvl) // Previous bar touched/pierced
                bool reject_bear = close < lvl and open < lvl // Current bar closes back below
                if alert_on_retest and retest_bear and reject_bear
                    alert("Retest and Rejection of " + txt + " at " + str.tostring(lvl), alert.freq_once_per_bar_close)

            // --- Delete line ---
            if delete_it
                line.delete(l_curr)
                label.delete(lbl_curr)
                if alert_on_invalidation
                    alert(txt + " at " + str.tostring(lvl) + " invalidated.", alert.freq_once_per_bar_close)
                array.remove(csd_lines, i)
                array.remove(csd_lbls, i)

    // HP-CISD Line Management (Same logic as CISD)
    if array.size(hp_csd_lines) > 0
        for i = array.size(hp_csd_lines) - 1 to 0
            line  l_curr   = array.get(hp_csd_lines, i)
            label lbl_curr = array.get(hp_csd_lbls, i)
            float lvl = line.get_y1(l_curr)
            string txt = label.get_text(lbl_curr)

            bool delete_it = false

            // Bullish HP (+)
            if str.contains(txt, "(+HP)")
                // --- Invalidation Logic ---
                bool body_below_0 = (close < lvl and open < lvl)
                bool body_below_1 = (close[1] < lvl and open[1] < lvl)
                bool body_below_2 = (close[2] < lvl and open[2] < lvl)

                if body_below_0
                    line.set_color(l_curr, csd_pending_color)

                if body_below_0 and body_below_1 and body_below_2
                    delete_it := true

                if close > lvl or open > lvl
                    line.set_color(l_curr, hp_csd_bull_clr)

                bool retest_bull = low[1] <= lvl and (close[1] < lvl or open[1] < lvl)
                bool reject_bull = close > lvl and open > lvl
                if alert_on_retest and retest_bull and reject_bull
                    alert("Retest and Rejection of " + txt + " at " + str.tostring(lvl), alert.freq_once_per_bar_close)

            // Bearish HP (-)
            else if str.contains(txt, "(-HP)")
                // --- Invalidation Logic ---
                bool body_above_0 = (close > lvl and open > lvl)
                bool body_above_1 = (close[1] > lvl and open[1] > lvl)
                // bool body_above_2 = (close[2] > lvl and open[2] > lvl)

                if body_above_0
                    line.set_color(l_curr, csd_pending_color)

                if body_above_0 and body_above_1// and body_above_2
                    delete_it := true

                if close < lvl or open < lvl
                    line.set_color(l_curr, hp_csd_bear_clr)

                bool retest_bear = high[1] >= lvl and (close[1] > lvl or open[1] > lvl)
                bool reject_bear = close < lvl and open < lvl
                if alert_on_retest and retest_bear and reject_bear
                    alert("Retest and Rejection of " + txt + " at " + str.tostring(lvl), alert.freq_once_per_bar_close)

            if delete_it
                line.delete(l_curr)
                label.delete(lbl_curr)
                if alert_on_invalidation
                    alert(txt + " at " + str.tostring(lvl) + " invalidated.", alert.freq_once_per_bar_close)
                array.remove(hp_csd_lines, i)
                array.remove(hp_csd_lbls, i)
    _max_show

// 3. Execution
f_run_csd(settings.buffer, max_csd_show)
